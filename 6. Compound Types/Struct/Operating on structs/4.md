struct Person {

solution code->
struct Person {
    name: String,
    age: u8,
}
fn main() {
    let age = 18;
    let mut p = Person {
        name: String::from("sunface"),
        age,
    };

    // how can you believe sunface is only 18? 
    p.age = 30;

    p.name = String::from("sunfei");
    println!("Success!");
}

output-> Success!

blanks filled-> p.name is used to add string sunfei
